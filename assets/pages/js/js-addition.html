JavaScript Example

Below is an example of JavaScript code that interacts with the user:

function getName() {
    let name;
    while (true) {
        name = prompt("What is your name?");
        if (!name) {
            alert("Name cannot be empty. Please enter a valid name.");
            continue;
        }
        // Check if the input contains only letters
        if (/^[a-zA-Z]+$/.test(name)) {
            break;
        } else {
            alert("Invalid input. Please use only letters.");
        }
    }
    // Capitalize the first letter
    name = name.charAt(0).toUpperCase() + name.slice(1).toLowerCase();
    return name;
}

function getNumber(promptText) {
    let num;
    while (true) {
        num = prompt(promptText);
        // Check if the input is a valid number
        if (!isNaN(num) && num.trim() !== "") {
            return Number(num);
        } else {
            alert("Invalid input. Please enter a valid number.");
        }
    }
}
